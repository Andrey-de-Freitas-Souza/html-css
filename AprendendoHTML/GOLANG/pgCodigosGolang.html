Declaração de variáveis

    package main

    import (
        "fmt"
    )
    
    // variavel que funciona em todo package
    var w = "Variavel package level escope"
    
    func main() {
        /*variaveis que funcionam apenas nesse codeblock e
        apenas se forem declaradas antes de serem chamadas*/
        x := 10
        y := "Olá, Bom dia!"
        z := false
    
        //utilizado o comando Printf (print formatado)
        //%v = Valor
        //%T = Tipo
        fmt.Printf("x: , %v, %T\n", x, x)
        fmt.Printf("y: , %v, %T\n", y, y)
        fmt.Printf("z: , %v, %T\n", z, z)
    
        x, j := 20, 30
        fmt.Printf("x: , %v, %T\n", x, x)
        fmt.Printf("j: , %v, %T\n", j, j)
        fmt.Printf("w: , %v, %T\n", w, w)
    
    }

Resultado:

        x: , 10, int
        y: , Olá, Bom dia!, string
        z: , false, bool
        x: , 20, int
        j: , 30, int
        w: , Variavel package level escope, string
